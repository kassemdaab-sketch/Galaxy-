function launch() { const input = document.getElementById('urlInput').value.trim(); if (!input) return; let targetUrl = ""; if (input.includes('.') && !input.includes(' ')) { targetUrl = input.startsWith('http') ? input : 'https://' + input; } else { // We are using the standard search URL now because the proxy will handle the rest targetUrl = `https://duckduckgo.com/?q=${encodeURIComponent(input)}`; } // This MUST match the way your proxy reads the "url" variable const proxyPath = `/api/proxy?url=${encodeURIComponent(targetUrl)}`; document.getElementById('content-frame').src = proxyPath; }
