function launch() { const input = document.getElementById('urlInput').value.trim(); if (!input) return; let targetUrl = ""; if (input.includes('.') && !input.includes(' ')) { targetUrl = input.startsWith('http') ? input : 'https://' + input; } else { // This ensures the search results load properly in Node.js targetUrl = `https://duckduckgo.com/html/?q=${encodeURIComponent(input)}`; } const proxyPath = `/api/proxy?url=${encodeURIComponent(targetUrl)}`; document.getElementById('content-frame').src = proxyPath; }
