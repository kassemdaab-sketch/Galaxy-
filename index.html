function launchProxy(val) { if(!val) return; const frame = document.getElementById('proxy-frame'); const viewport = document.getElementById('proxy-layer'); let targetUrl = val.trim(); // If it's not a link, search DuckDuckGo if(!targetUrl.includes('.') || targetUrl.includes(' ')) { targetUrl = 'https://duckduckgo.com/?q=' + encodeURIComponent(targetUrl); } else { if(!targetUrl.startsWith('http')) targetUrl = 'https://' + targetUrl; } // Sends the request through your Node.js "Bypass" route frame.src = window.location.origin + '/proxy/' + targetUrl; viewport.style.display = 'flex'; }
