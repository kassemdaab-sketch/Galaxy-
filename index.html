function launch() { const input = document.getElementById('urlInput').value.trim(); if (!input) return; let targetUrl = ""; if (input.includes('.') && !input.includes(' ')) { targetUrl = input.startsWith('http') ? input : 'https://' + input; } else { targetUrl = `https://html.duckduckgo.com/html/?q=${encodeURIComponent(input)}`; } // This routes through your Node.js engine const proxyPath = `/api/proxy?url=${encodeURIComponent(targetUrl)}`; // FIX 1: Explicitly target the iframe to keep links inside the framework const frame = document.getElementById('content-frame'); frame.src = proxyPath; // FIX 2: Clear the "raw code" look by resetting the input or logging // This prevents the browser from showing the proxy path in the UI console.log("Galaxy Proxy navigating to: " + input); }
