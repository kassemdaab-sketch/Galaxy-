const express = require('express'); const { createProxyMiddleware } = require('http-proxy-middleware'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // This serves your HTML files app.use(express.static(path.join(__dirname, 'public'))); // The Bypass Engine: Strips security headers so sites work in the frame app.use('/proxy', createProxyMiddleware({ target: 'http://localhost:3000', // Placeholder router: (req) => req.url.split('/proxy/')[1], changeOrigin: true, pathRewrite: (path, req) => '', onProxyRes: function (proxyRes) { // This is the "Bypass" part: it deletes the blocks delete proxyRes.headers['x-frame-options']; delete proxyRes.headers['content-security-policy']; } })); app.listen(PORT, () => console.log(`Galaxy Engine spinning on ${PORT}`));
