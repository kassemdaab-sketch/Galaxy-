const express = require('express'); const { createProxyMiddleware } = require('http-proxy-middleware'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Serves your Galaxy HTML files from the 'public' folder app.use(express.static(path.join(__dirname, 'public'))); app.use('/proxy', createProxyMiddleware({ target: 'http://localhost:3000', router: (req) => { // Extracts the URL after /proxy/ const url = req.url.split('/proxy/')[1]; return url.startsWith('http') ? url : 'https://' + url; }, changeOrigin: true, pathRewrite: (path, req) => '', onProxyRes: function (proxyRes) { // Deletes the security locks so the site shows up in your Galaxy frame delete proxyRes.headers['x-frame-options']; delete proxyRes.headers['content-security-policy']; delete proxyRes.headers['cross-origin-resource-policy']; }, // Adding this helps with sites like Roblox that have strict cookie/origin rules onProxyReq: (proxyReq, req, res) => { proxyReq.setHeader('User-Agent', 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/119.0.0.0 Safari/537.36'); } })); app.listen(PORT, () => console.log(`Galaxy Engine spinning on ${PORT}`));
