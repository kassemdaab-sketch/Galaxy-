function launch() { const input = document.getElementById('urlInput').value.trim(); if (!input) return; let targetUrl = ""; // If it's a URL (has a dot and no spaces) if (input.includes('.') && !input.includes(' ')) { targetUrl = input.startsWith('http') ? input : 'https://' + input; } // If it's a search term, use the HTML version of DuckDuckGo else { // 'https://html.duckduckgo.com/html/' is the secret for proxy compatibility targetUrl = `https://html.duckduckgo.com/html/?q=${encodeURIComponent(input)}`; } // Connects to your Node.js engine const proxyPath = `/api/proxy?url=${encodeURIComponent(targetUrl)}`; document.getElementById('content-frame').src = proxyPath; }
